```python
class Solution(object):
    def rotate(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: None Do not return anything, modify nums in-place instead.
        """
        cp = nums[:]
        for i in range(len(nums)):
            nums[i]=cp[(len(nums)-k+i)%len(nums)]
```
```python
class Solution(object):
    def rotate(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: None Do not return anything, modify nums in-place instead.
        """
        l = len(nums)
        nums[:] = nums[-(k % l):] + nums[:-(k % l)]
```
